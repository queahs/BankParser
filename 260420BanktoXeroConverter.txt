let rows = [];
let detectedFormat = ''; // 'armalytix' | 'dext' | 'bank-direct'

// ─── Format detection ─────────────────────────────────────────────────────────
function detectFormat(headers) {
  const h = headers.map(x => x.toLowerCase().replace(/\*/g,''));
  if (h.includes('booking_date') || h.includes('merchantname') || h.includes('cleansed_description')) return 'armalytix';
  if (h.includes('date') && h.includes('description') && h.includes('amount')) return 'dext';
  return 'unknown';
}

// ─── Categorisation ───────────────────────────────────────────────────────────
function categorise(haystack) {
  const h = haystack.toLowerCase();
  for (const r of RULES) {
    if (r.kws.some(k => h.includes(k))) {
      return { account: r.account, tax: r.tax, cat: r.cat };
    }
  }
  return { account: '', tax: '', cat: '__unmatched__' };
}

// ─── CSV parsing ──────────────────────────────────────────────────────────────
function splitLine(line) {
  const vals = []; let cur = '', inQ = false;
  for (let i = 0; i < line.length; i++) {
    if (line[i] === '"') { inQ = !inQ; }
    else if (line[i] === ',' && !inQ) { vals.push(cur); cur = ''; }
    else { cur += line[i]; }
  }
  vals.push(cur);
  return vals.map(v => v.replace(/^"|"$/g, '').trim());
}

function parseCSV(text) {
  const lines = text.trim().split('\n').filter(l => l.trim());
  if (!lines.length) return { headers: [], records: [] };
  const rawHeaders = splitLine(lines[0]);
  // Normalise headers: strip leading *, trim whitespace, lowercase
  const headers = rawHeaders.map(h => h.trim().replace(/^\*/, '').toLowerCase());
  const records = lines.slice(1).map(line => {
    const vals = splitLine(line);
    const row = {};
    headers.forEach((h, i) => row[h] = vals[i] || '');
    return row;
  });
  return { headers, records };
}

// ─── File handling ────────────────────────────────────────────────────────────
function handleFile(file) {
  if (!file.name.toLowerCase().endsWith('.csv')) { showError('Please upload a .csv file.'); return; }
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const { headers, records } = parseCSV(e.target.result);
      if (!records.length) { showError('No data found in file.'); return; }

      const fmt = detectFormat(headers);
      if (fmt === 'unknown') { showError('Format not recognised. Please upload an Armalytix, Dext, or Bank Direct CSV export.'); return; }
      detectedFormat = fmt;

      if (fmt === 'armalytix') {
        rows = parseArmalytix(records);
      } else {
        rows = parseDext(records);
      }

      rows = rows.filter(r => r.cat !== '__payment__');
      if (!rows.length) { showError('No transactions found — check the file format.'); return; }

      populateCatFilter();
      document.getElementById('s-file').textContent = file.name;

      const badge = document.getElementById('format-badge-display');
      const badgeLabels = { armalytix: '⬡ Armalytix', dext: '⬡ Dext', 'bank-direct': '⬡ Bank Direct' };
      const badgeClasses = { armalytix: 'armalytix', dext: 'dext', 'bank-direct': 'dext' };
      badge.innerHTML = '<span class="format-badge ' + (badgeClasses[fmt]||'dext') + '">' + (badgeLabels[fmt]||fmt) + '</span>';

      document.getElementById('upload-section').style.display = 'none';
      document.getElementById('result-section').style.display = 'block';
      renderTable();
    } catch(err) {
      showError('Could not parse file. Please check it is a valid Armalytix or Dext CSV export.');
    }
  };
  reader.readAsText(file);
}

function parseArmalytix(records) {
  return records
    .filter(r => r.booking_date || r.amount)
    .map((r, idx) => {
      const merchant = (r.merchantname || '').trim();
      const cleansed = (r.cleansed_description || '').trim();
      const desc     = (r.description || '').trim();
      const payee    = merchant || cleansed || desc;
      const haystack = payee + ' ' + (cleansed || desc);
      const c = categorise(haystack);
      return {
        idx,
        date:    (r.booking_date || '').trim(),
        amount:  (parseFloat(r.amount) || 0).toFixed(2),
        desc:    payee,
        ref:     (r.reference || '').trim(),
        account: c.account,
        tax:     c.tax,
        cat:     c.cat,
      };
    });
}

function parseDext(records) {
  return records
    .filter(r => r.date || r.amount)
    .map((r, idx) => {
      const desc = (r.description || '').trim();
      const c = categorise(desc);
      return {
        idx,
        date:    (r.date || '').trim(),
        amount:  (r.amount || '').trim(),
        desc,
        ref:     (r.reference || r['cheque number'] || '').trim(),
        account: c.account,
        tax:     c.tax,
        cat:     c.cat,
      };
    });
}

function learnFromFile(file) {
  if (!file.name.toLowerCase().endsWith('.csv')) { showLearnMsg('Please upload a .csv file.', true); return; }
  const reader = new FileReader();
  reader.onload = e => {
    try {
      const { records } = parseCSV(e.target.result);
      let added = 0, updated = 0;
      const seen = new Set();
      records.forEach(r => {
        const payee = (r['payee'] || '').trim();
        const code  = (r['accountcode'] || '').trim();
        const tax   = (r['taxtype'] || '').trim();
        if (!payee || !code || !tax) return;
        const key = payee.toLowerCase();
        if (seen.has(key)) return;
        seen.add(key);

        const existIdx = RULES.findIndex(rule => rule.kws.includes(key));
        if (existIdx >= 0) {
          if (RULES[existIdx].account !== code || RULES[existIdx].tax !== tax) {
            RULES[existIdx].account = code; RULES[existIdx].tax = tax; updated++;
          }
        } else {
          RULES.unshift({ kws:[key], account:code, tax:tax, cat:payee, learned:true });
          added++;
        }
      });

      const parts = [];
      if (added)   parts.push(added + ' new rule' + (added > 1 ? 's' : '') + ' added');
      if (updated) parts.push(updated + ' rule' + (updated > 1 ? 's' : '') + ' updated');
      if (!parts.length) parts.push('No changes — rules already up to date');
      showLearnMsg('✓ ' + parts.join(', ') + '.', false);
      if (rows.length) { reapplyRules(); renderTable(); updateStats(); }
    } catch(err) {
      showLearnMsg('Could not read file. Make sure it has Payee, *AccountCode and *TaxType columns.', true);
    }
  };
  reader.readAsText(file);
}

function showLearnMsg(msg, isErr) {
  learnMsg.textContent = msg;
  learnMsg.className = 'learn-msg ' + (isErr ? 'err' : 'ok');
  learnMsg.style.display = 'block';
}

function reapplyRules() {
  rows.forEach(r => {
    const c = categorise(r.desc);
    if (r.cat === '__unmatched__') { r.account = c.account; r.tax = c.tax; r.cat = c.cat; }
  });
}

// ─── Sort state ───────────────────────────────────────────────────────────────
let sortCol = 'date';
let sortDir = 'asc';

function handleSortClick(th) {
  const col = th.dataset.col;
  if (sortCol === col) {
    sortDir = sortDir === 'asc' ? 'desc' : 'asc';
  } else {
    sortCol = col;
    sortDir = 'asc';
  }
  renderTable();
}

function updateSortHeaders() {
  document.querySelectorAll('#tbl-head th').forEach(th => {
    th.classList.remove('sort-asc', 'sort-desc');
    const icon = th.querySelector('.sort-icon');
    if (!icon) return;
    if (th.dataset.col === sortCol) {
      th.classList.add(sortDir === 'asc' ? 'sort-asc' : 'sort-desc');
      icon.textContent = sortDir === 'asc' ? '↑' : '↓';
    } else {
      icon.textContent = '↕';
    }
  });
}

// ─── Table rendering ──────────────────────────────────────────────────────────
function populateCatFilter() {
  const sel = document.getElementById('filter-cat');
  while (sel.options.length > 2) sel.remove(2);
  const cats = [...new Set(rows.map(r => r.cat).filter(c => c !== '__unmatched__' && c !== '__payment__'))].sort();
  cats.forEach(c => { const o = document.createElement('option'); o.value = c; o.textContent = c; sel.appendChild(o); });
}

function renderTable() {
  const catFilter = document.getElementById('filter-cat').value;
  const search    = document.getElementById('filter-search').value.toLowerCase();
  const tbody     = document.getElementById('tbl-body');
  tbody.innerHTML = '';

  const filtered = rows.filter(r => {
    if (catFilter === '__unmatched__' && r.cat !== '__unmatched__') return false;
    if (catFilter && catFilter !== '__unmatched__' && r.cat !== catFilter) return false;
    if (search && !r.desc.toLowerCase().includes(search) && !r.date.includes(search)) return false;
    return true;
  });

  document.getElementById('showing-count').textContent = filtered.length + ' of ' + rows.length + ' rows';

  // Sort
  filtered.sort((a, b) => {
    let va = a[sortCol] || '', vb = b[sortCol] || '';
    if (sortCol === 'amount') {
      va = parseFloat(va) || 0; vb = parseFloat(vb) || 0;
      return sortDir === 'asc' ? va - vb : vb - va;
    }
    // Date: parse DD/MM/YYYY
    if (sortCol === 'date') {
      const pd = s => { const p = s.split('/'); return p.length===3 ? new Date(p[2],p[1]-1,p[0]) : new Date(s); };
      const da = pd(va), db = pd(vb);
      return sortDir === 'asc' ? da - db : db - da;
    }
    va = va.toLowerCase(); vb = vb.toLowerCase();
    return sortDir === 'asc' ? va.localeCompare(vb) : vb.localeCompare(va);
  });

  updateSortHeaders();

  filtered.forEach(r => {
    const tr   = document.createElement('tr');
    const amt  = parseFloat(r.amount);
    const amtC = isNaN(amt) ? '' : (amt >= 0 ? 'amt-pos' : 'amt-neg');
    const amtS = isNaN(amt) ? r.amount : ((amt >= 0 ? '+' : '') + amt.toFixed(2));

    const acctSel = buildSelect('account', r.idx, r.account,
      [['', '— select —'], ...ACCOUNTS.map(a => [a.code, a.code + ' — ' + a.name])]);
    const taxSel  = buildSelect('tax', r.idx, r.tax,
      [['', '— select —'], ...TAX_TYPES.map(t => [t, t])]);

    tr.innerHTML =
      '<td title="'+esc(r.date)+'">'+esc(r.date)+'</td>' +
      '<td class="'+amtC+'">'+amtS+'</td>' +
      '<td title="'+esc(r.desc)+'">'+esc(r.desc)+'</td>' +
      '<td></td><td></td>';

    tr.children[3].appendChild(acctSel);
    tr.children[4].appendChild(taxSel);
    tbody.appendChild(tr);
  });

  updateStats();
}

function buildSelect(field, idx, current, options) {
  const sel = document.createElement('select');
  sel.dataset.idx   = idx;
  sel.dataset.field = field;
  if (!current) sel.className = 'unset';
  sel.onchange = () => { updateRow(idx, field, sel.value); sel.className = sel.value ? '' : 'unset'; };
  options.forEach(([val, label]) => {
    const o = document.createElement('option');
    o.value = val; o.textContent = label;
    if (val === current) o.selected = true;
    sel.appendChild(o);
  });
  return sel;
}

// Tracks rows the user has manually edited — these won't be overwritten by propagation
const manuallyEdited = new Set();

// Auto-tax defaults by account code
const ACCOUNT_TAX_DEFAULTS = {
  '301': 'Zero Rated Expenses',
  '999': 'No VAT',
  '820': 'No VAT',
  '600': 'No VAT',
  '700': 'No VAT',
  '202': 'Exempt Expenses',
};

function updateRow(idx, field, value) {
  const r = rows.find(r => r.idx === idx);
  if (!r) return;

  if (field === 'account') r.account = value;
  if (field === 'tax')     r.tax     = value;

  // Auto-set tax type when a known account code is chosen (only if tax not yet set by user)
  if (field === 'account' && ACCOUNT_TAX_DEFAULTS[value]) {
    const autoTax = ACCOUNT_TAX_DEFAULTS[value];
    r.tax = autoTax;
    const taxSel = document.querySelector('select[data-idx="'+idx+'"][data-field="tax"]');
    if (taxSel) { taxSel.value = autoTax; taxSel.className = ''; }
  }

  if (r.account && r.tax && r.cat === '__unmatched__') r.cat = 'manual';

  // ── Propagate to siblings — but only if THIS row hasn't been individually overridden ──
  // Mark this row as having been touched
  const wasAlreadyEdited = manuallyEdited.has(idx);
  manuallyEdited.add(idx);

  const descNorm = r.desc.trim().toLowerCase();
  // Siblings = same description, NOT already individually overridden by user
  const siblings = rows.filter(s =>
    s.idx !== idx &&
    s.desc.trim().toLowerCase() === descNorm &&
    !manuallyEdited.has(s.idx)   // skip rows user has already manually set
  );

  if (!wasAlreadyEdited && siblings.length > 0) {
    // First edit on this row — propagate to unedited siblings
    siblings.forEach(s => {
      if (field === 'account') {
        s.account = value;
        if (ACCOUNT_TAX_DEFAULTS[value]) {
          s.tax = ACCOUNT_TAX_DEFAULTS[value];
        }
      }
      if (field === 'tax') s.tax = value;
      if (s.account && s.tax && s.cat === '__unmatched__') s.cat = 'manual';

      const aSel = document.querySelector('select[data-idx="'+s.idx+'"][data-field="account"]');
      const tSel = document.querySelector('select[data-idx="'+s.idx+'"][data-field="tax"]');
      if (aSel) { aSel.value = s.account; aSel.className = s.account ? '' : 'unset'; }
      if (tSel) { tSel.value = s.tax;     tSel.className = s.tax     ? '' : 'unset'; }
    });

    showToast('Applied to ' + siblings.length + ' other "' + r.desc + '" row' + (siblings.length > 1 ? 's' : '') + '. Further changes on this row won\'t affect the others.');
  } else if (wasAlreadyEdited) {
    // Subsequent edit — only affects this row
    showToast('"' + r.desc + '" updated — other rows unchanged.');
  }

  updateStats();
}

// ── Toast notification ────────────────────────────────────────────────────────
let toastTimer = null;
function showToast(msg) {
  let toast = document.getElementById('propagate-toast');
  if (!toast) {
    toast = document.createElement('div');
    toast.id = 'propagate-toast';
    toast.style.cssText = [
      'position:fixed','bottom:24px','left:50%','transform:translateX(-50%)',
      'background:#1a1916','color:#fff','font-family:var(--sans)','font-size:12px',
      'padding:10px 18px','border-radius:8px','box-shadow:0 4px 16px rgba(0,0,0,0.25)',
      'z-index:9999','pointer-events:none','opacity:0',
      'transition:opacity 0.2s ease','max-width:90vw','text-align:center'
    ].join(';');
    document.body.appendChild(toast);
  }
  toast.textContent = msg;
  toast.style.opacity = '1';
  clearTimeout(toastTimer);
  toastTimer = setTimeout(() => { toast.style.opacity = '0'; }, 3500);
}

function updateStats() {
  const unmatched = rows.filter(r => !r.account || !r.tax).length;
  const matched   = rows.length - unmatched;
  document.getElementById('s-total').textContent     = rows.length;
  document.getElementById('s-matched').textContent   = matched;
  document.getElementById('s-unmatched').textContent = unmatched;
  const wb = document.getElementById('warn-box');
  if (unmatched > 0) {
    wb.style.display = 'block';
    wb.textContent = unmatched + ' transaction' + (unmatched !== 1 ? 's' : '') + ' still need an account code and/or tax type. Use the "Needs review" filter to find them.';
  } else {
    wb.style.display = 'none';
  }
}

      .map(v => '"'+String(v||'').replace(/"/g,'""')+'"').join(','));

  const a    = document.createElement('a');
  a.href     = URL.createObjectURL(blob);
  a.download = 'xero_import.csv';
  a.click();
}
